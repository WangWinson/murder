    Docs

    The docs task builds docco files, switches to the gh-pages branch, commits the docs, and switches back to the development branch.

    Usage: gulp docs

    gulp.task('docs', function () {

  'cp -r docs/ docscopy/',
  'git checkout gh-pages',
  'rm -rf docs/',
  'mv docscopy/ docs/',
  'git add docs/',
  'git commit -m \"updates docs\"',
  'git checkout master'
]));

gulp.task('docs-push', function() {
  git.push('origin', 'gh-pages');

    gulp.task('test-dev', function (done) {
  startKarma('dev', done);
});

function startKarma(name, done) {
  exampleServer.listen(function () {
    karma.start({
      configFile: __dirname + '/karma/' + (name ? name + '.' : '') + 'conf.js'
    }, function (karmaExitCode) {
      done(karmaExitCode);
      setTimeout(process.exit.bind(process, karmaExitCode), 5000);
    });
  });
}
